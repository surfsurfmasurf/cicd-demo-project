Abstract Flower

이 작품은 수학적 함수를 이용해 생성된 추상적인 꽃 모양의 애니메이션입니다. sin, cos, mag 함수를 복합적으로 사용하여 유기적인 움직임을 만들어냅니다.

Code

let t = 0;

function setup() {
  let canvas = createCanvas(400, 400);
  canvas.parent('canvas-container');
}

function draw() {
  background(9);
  stroke(400, 116);
  t += PI / 90;
  for (let i = 10000; i--; ) {
    let y = i / 790;
    let k = (y < 8 ? 9 + sin(y ^ 9) * 6 : 4 + cos(y)) * cos(i + t / 4);
    let e = y / 3 - 13;
    let d = mag(k, e) + cos(e + t * 2 + (i % 2) * 4);
    let q = (y * k / 5) * (2 + sin(d * 2 + y - t * 4)) + 80;
    let c = d / 4 - t / 2 + (i % 2) * 3;
    point(q * cos(c) + 200, q * sin(c) + d * 9 + 60);
  }
}
