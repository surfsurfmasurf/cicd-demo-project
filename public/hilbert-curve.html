// Hilbert Curve Mapping Logic
function hilbert(i, order) {
    let points = [
        createVector(0, 0), createVector(0, 1),
        createVector(1, 1), createVector(1, 0)
    ];
    let index = i & 3;
    let v = points[index];
    
    for (let j = 1; j < order; j++) {
        i = i >>> 2;
        index = i & 3;
        let len = pow(2, j);
        if (index == 0) {
            let temp = v.x; v.x = v.y; v.y = temp;
        } else if (index == 1) {
            v.y += len;
        } else if (index == 2) {
            v.x += len; v.y += len;
        } else if (index == 3) {
            let temp = len - 1 - v.x;
            v.x = len - 1 - v.y;
            v.y = temp;
            v.x += len;
        }
    }
    return v;
}
